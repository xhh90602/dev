### 防抖
``` javascript
const debounce = (fn, delay) => {
    let time = null;
    return (...args) => {
        clearTimeout(time);
        time = setTimeout(() => {
            fn.call(this, args);
        }, delay);

    }
}

```

### 节流
``` javascript
const throttle = (fn, delay) => {
    let flag = false;
    return (...args) => {
        if (!flag) {
            flag = true;
            fn.call(this, args);
            setTimeout(() => {
                flag = false;
            }, delay);
        }
    }
}


const throttle = (fn, delay) => {
    let time = 0;
    return (...args) => {
        if (Date.now() - time > delay) {
            time = Date.now();
            fn.call(this, args);
        }
    }
}
```
